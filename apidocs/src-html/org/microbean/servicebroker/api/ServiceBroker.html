017package org.microbean.servicebroker.api;
018
019import javax.validation.constraints.NotNull;
020
021import org.microbean.servicebroker.api.query.LastOperationQuery;
022
023import org.microbean.servicebroker.api.query.state.Catalog;
024import org.microbean.servicebroker.api.query.state.LastOperation;
025
026import org.microbean.servicebroker.api.command.DeleteBindingCommand;
027import org.microbean.servicebroker.api.command.DeleteServiceInstanceCommand;
028import org.microbean.servicebroker.api.command.ProvisionBindingCommand;
029import org.microbean.servicebroker.api.command.ProvisionServiceInstanceCommand;
030import org.microbean.servicebroker.api.command.UpdateServiceInstanceCommand;
031
032public abstract class ServiceBroker {
033
034  @NotNull
035  public LastOperation getLastOperation(@NotNull final LastOperationQuery lastOperationQuery) throws ServiceBrokerException {
036    throw new ServiceBrokerException(new UnsupportedOperationException());
037  }
038
039  public boolean isSupportedServiceId(final String serviceId) throws ServiceBrokerException {
040    return serviceId != null;
041  }
042
043  public boolean isSupportedPlanId(final String serviceId, final String planId) throws ServiceBrokerException {
044    return planId != null;
045  }
046
047  public boolean isLive() throws ServiceBrokerException {
048    return true;
049  }
050
051  public boolean isReady() throws ServiceBrokerException {
052    return true;
053  }
054  
055  public abstract boolean isPlanBindable(final String serviceId, final String planId) throws ServiceBrokerException;
056  
057  public boolean isAsynchronousOnly() {
058    return false;
059  }
060  
061  @NotNull
062  public abstract Catalog getCatalog() throws ServiceBrokerException;
063
064  @NotNull
065  public abstract ProvisionBindingCommand.Response execute(@NotNull final ProvisionBindingCommand command) throws ServiceBrokerException;
066
067  @NotNull
068  public abstract DeleteBindingCommand.Response execute(@NotNull final DeleteBindingCommand command) throws ServiceBrokerException;
069
070  @NotNull
071  public abstract ProvisionServiceInstanceCommand.Response execute(@NotNull final ProvisionServiceInstanceCommand command) throws ServiceBrokerException;
072
073  @NotNull
074  public abstract UpdateServiceInstanceCommand.Response execute(@NotNull final UpdateServiceInstanceCommand command) throws ServiceBrokerException;
075
076  @NotNull
077  public abstract DeleteServiceInstanceCommand.Response execute(@NotNull final DeleteServiceInstanceCommand command) throws ServiceBrokerException;
078
079}
