113  public CompositeServiceBroker(final Set<? extends ServiceBroker> serviceBrokers, final boolean parallelServiceDiscovery) {
114    super();
115    this.serviceBrokersLock = new ReentrantReadWriteLock();
116    this.serviceBrokerAssociationLock = new ReentrantReadWriteLock();
117    this.parallelServiceDiscovery = parallelServiceDiscovery;
118    this.serviceBrokers = new HashSet<>();
119    this.serviceBrokersByServiceId = new HashMap<>();
120    if (serviceBrokers != null) {
121      for (final ServiceBroker serviceBroker : serviceBrokers) {
122        if (serviceBroker != null) {
123          this.addServiceBroker(serviceBroker);
124        }
125      }
126    }
127  }

135  @Override
136  public boolean isLive() throws ServiceBrokerException {
137    boolean returnValue = true;
138    final Collection<? extends ServiceBroker> serviceBrokers = this.getServiceBrokers();
139    if (serviceBrokers == null || serviceBrokers.isEmpty()) {
140      returnValue = false;
141    } else {
142      for (final ServiceBroker serviceBroker : serviceBrokers) {
143        if (serviceBroker != null) {
144          if (!serviceBroker.isLive()) {
145            returnValue = false;
146            break;
147          }
148        }
149      }
150    }
151    return returnValue;
152  }

154  @Override
155  public boolean isReady() throws ServiceBrokerException {
156    boolean returnValue = true;
157    final Collection<? extends ServiceBroker> serviceBrokers = this.getServiceBrokers();
158    if (serviceBrokers == null || serviceBrokers.isEmpty()) {
159      returnValue = false;
160    } else {
161      for (final ServiceBroker serviceBroker : serviceBrokers) {
162        if (serviceBroker != null) {
163          if (!serviceBroker.isReady()) {
164            returnValue = false;
165            break;
166          }
167        }
168      }
169    }
170    return returnValue;
171  }

217  /**
218   * Calls the {@link #handleAddServiceBroker(ServiceBroker)} method
219   * supplying it the supplied {@code serviceBroker} and returns the
220   * result.
221   *
222   * <p>This method acquires and properly releases a write lock
223   * internally that ensures that overrides of the {@link
224   * #handleAddServiceBroker(ServiceBroker)} method do not have to
225   * worry about concurrency concerns, particularly while a {@link
226   * #getCatalog()} invocation is in process.</p>
227   *
228   * @param serviceBroker the {@link ServiceBroker} to add; must not
229   * be {@code null}
230   *
231   * @return {@code true} if the supplied {@link ServiceBroker} was
232   * added; {@code false} otherwise
233   *
234   * @exception NullPointerException if {@code serviceBroker} was
235   * {@code null}
236   *
237   * @see #handleAddServiceBroker(ServiceBroker)
238   */
239  public final boolean addServiceBroker(@NotNull final ServiceBroker serviceBroker) {
240    Objects.requireNonNull(serviceBroker);
241    if (serviceBroker == this) {
242      throw new IllegalArgumentException("serviceBroker == this");
243    }
244    try {
245      this.serviceBrokersLock.writeLock().lock();
246      return this.handleAddServiceBroker(serviceBroker);
247    } finally {
248      this.serviceBrokersLock.writeLock().unlock();
249    }
250  }
251
252  protected boolean handleAddServiceBroker(@NotNull final ServiceBroker serviceBroker) {
253    Objects.requireNonNull(serviceBroker);
254    if (serviceBroker == this) {
255      throw new IllegalArgumentException("serviceBroker == this");
256    }
257    return this.serviceBrokers.add(serviceBroker);
258  }

260  /**
261   * Calls the {@link #handleRemoveServiceBroker(ServiceBroker)} method
262   * supplying it the supplied {@code serviceBroker} and returns the
263   * result.
264   *
265   * <p>This method first calls the {@link
266   * #removeServices(ServiceBroker)} method supplying
267   * it the {@link ServiceBroker} that has just been removed, also
268   * with proper concurrency semantics.</p>
269   *
270   * <p>This method acquires and properly releases a write lock
271   * internally that ensures that overrides of the {@link
272   * #handleRemoveServiceBroker(ServiceBroker)} method do not have to
273   * worry about concurrency concerns, particularly while a {@link
274   * #getCatalog()} invocation is in process.</p>
275   *
276   * @param serviceBroker the {@link ServiceBroker} to remove; must not
277   * be {@code null}
278   *
279   * @return {@code true} if the supplied {@link ServiceBroker} was
280   * removeed; {@code false} otherwise
281   *
282   * @exception NullPointerException if {@code serviceBroker} was
283   * {@code null}
284   *
285   * @see #handleRemoveServiceBroker(ServiceBroker)
286   *
287   * @see #removeServices(ServiceBroker)
288   */
289  public final boolean removeServiceBroker(@NotNull final ServiceBroker serviceBroker) {
290    Objects.requireNonNull(serviceBroker);    
291    if (serviceBroker == this) {
292      throw new IllegalArgumentException("serviceBroker == this");
293    }
294    try {
295      this.serviceBrokersLock.writeLock().lock();
296      try {
297        this.serviceBrokerAssociationLock.writeLock().lock();
298        this.removeServices(serviceBroker);
299      } finally {
300        this.serviceBrokerAssociationLock.writeLock().unlock();
301      }
302      return this.handleRemoveServiceBroker(serviceBroker);
303    } finally {
304      this.serviceBrokersLock.writeLock().unlock();
305    }
306  }
307
308  protected boolean handleRemoveServiceBroker(@NotNull final ServiceBroker serviceBroker) {
309    Objects.requireNonNull(serviceBroker);
310    if (serviceBroker == this) {
311      throw new IllegalArgumentException("serviceBroker == this");
312    }
313    return this.serviceBrokers.remove(serviceBroker);
314  }

316  public final Set<ServiceBroker> getServiceBrokers() {
317    try {
318      this.serviceBrokersLock.readLock().lock();
319      return this.handleGetServiceBrokers();
320    } finally {
321      this.serviceBrokersLock.readLock().unlock();
322    }
323  }

378  /**
379   * Effectively emoves all {@link Service}s associated with the
380   * supplied {@link ServiceBroker} from being visible or known about
381   * in any way by this {@link CompositeServiceBroker} during an
382   * invocation of the {@link #removeServiceBroker(ServiceBroker)}
383   * method immediately before the supplied {@link ServiceBroker} is
384   * removed from this {@link CompositeServiceBroker} by the {@link
385   * #handleRemoveServiceBroker(ServiceBroker)} method.
386   *
387   * <p>Overrides of this method must not call the following methods,
388   * or undefined behavior may result:</p>
389   *
390   * <ul>
391   *
392   * <li>{@link #removeServiceBroker(ServiceBroker)}</li>
393   *
394   * <li>{@link #handleAddServiceBroker(ServiceBroker)}</li>
395   *
396   * <li>{@link #handleRemoveServiceBroker(ServiceBroker)}</li>
397   *
398   * <li>{@link #putServiceBrokerForServiceId(String,
399   * ServiceBroker)}</li>
400   *
401   * </ul>
402   *
403   * @param serviceBroker the {@link ServiceBroker} whose logical
404   * {@link Service} entries should be logically removed from this
405   * {@link CompositeServiceBroker}; must not be {@code null}
406   *
407   * @exception NullPointerException if {@code serviceBroker} is {@code null}
408   *
409   * @exception IllegalArgumentException if {@code serviceBroker} is
410   * this {@link CompositeServiceBroker}
411   */
412  protected void removeServices(@NotNull final ServiceBroker serviceBroker) {
413    Objects.requireNonNull(serviceBroker);
414    if (serviceBroker == this) {
415      throw new IllegalArgumentException("serviceBroker == this");
416    }
417    final Iterable<Entry<String, ServiceBroker>> entrySet = this.serviceBrokersByServiceId.entrySet();
418    if (entrySet != null) {
419      final Iterator<Entry<String, ServiceBroker>> iterator = entrySet.iterator();
420      if (iterator != null && iterator.hasNext()) {
421        while (iterator.hasNext()) {
422          final Entry<String, ServiceBroker> entry = iterator.next();
423          if (entry != null && serviceBroker.equals(entry.getValue())) {
424            iterator.remove();
425          }
426        }
427      }
428    }
429      
430  }

494  @Override
495  public boolean isPlanBindable(final String serviceId, final String planId) throws ServiceBrokerException {
496    Objects.requireNonNull(serviceId);
497    Objects.requireNonNull(planId);
498
499    boolean returnValue = false;
500    
501    ServiceBroker serviceBroker = null;
502    try {
503      this.serviceBrokerAssociationLock.readLock().lock();
504      serviceBroker = this.getServiceBrokerForServiceId(serviceId);
505    } finally {
506      this.serviceBrokerAssociationLock.readLock().unlock();
507    }
508    if (serviceBroker != null) {
509      returnValue = serviceBroker.isPlanBindable(serviceId, planId);
510    }
511    
512    return returnValue;
513  }

554  @NotNull
555  @Override
556  public final LastOperation getLastOperation(final LastOperationQuery lastOperationQuery) throws ServiceBrokerException {
557    Objects.requireNonNull(lastOperationQuery);    
558    LastOperation returnValue = null;
559    final String serviceId = lastOperationQuery.getServiceId();
560    if (serviceId != null) {
561      ServiceBroker serviceBroker = null;
562      try {
563        this.serviceBrokerAssociationLock.readLock().lock();
564        serviceBroker = this.getServiceBrokerForServiceId(serviceId);
565      } finally {
566        this.serviceBrokerAssociationLock.readLock().unlock();
567      }
568      if (serviceBroker != null) {
569        returnValue = serviceBroker.getLastOperation(lastOperationQuery);
570      }
571    }
572    if (returnValue == null) {
573      throw new InvalidServiceBrokerQueryException(lastOperationQuery);
574    }
575    return returnValue;
576  }

578  /**
579   * Returns a {@link Catalog} that represents the logical union of
580   * all the {@link Catalog} instances {@linkplain
581   * ServiceBroker#getCatalog() from all the
582   * <code>ServiceBroker</code>s} that are multiplexed by this {@link
583   * CompositeServiceBroker}.
584   *
585   * <p>This method never returns {@code null}.</p>
586   *
587   * <p>This method is safe for concurrent use by multiple
588   * threads.</p>
589   *
590   * <p>The implementation of this method is governed by the return
591   * value of the {@link #getParallelServiceDiscovery()} method: if
592   * {@code true}, then the {@link ServiceBroker}s that this {@link
593   * CompositeServiceBroker} multiplexes are consulted for their
594   * {@link Catalog} instances in parallel.</p>
595   *
596   * <p>Throughout the implementation of this method, appropriate
597   * locking semantics are used such that overrides of the various
598   * {@code protected} methods in this class do not have to
599   * synchronize or otherwise guard against concurrent access.</p>
600   *
601   * @return a non-{@code null} {@link Catalog}
602   *
603   * @exception ServiceBrokerException if an error occurs
604   *
605   * @see #getServiceBrokers()
606   *
607   * @see #getParallelServiceDiscovery()
608   *
609   * @see #getCatalog(ServiceBroker)
610   *
611   * @see #putServiceBrokerForServiceId(String, ServiceBroker)
612   *
613   * @see #removeServices()
614   */
615  @NotNull
616  @Override
617  public final Catalog getCatalog() throws ServiceBrokerException {
618    Catalog returnValue = null;
619    Set<ServiceBroker> serviceBrokers = null;
620    try {
621      this.serviceBrokersLock.readLock().lock();
622      serviceBrokers = this.getServiceBrokers();
623      if (serviceBrokers == null || serviceBrokers.isEmpty()) {
624        serviceBrokers = Collections.emptySet();
625      } else {
626        serviceBrokers = new HashSet<>(serviceBrokers);
627      }
628    } finally {
629      this.serviceBrokersLock.readLock().unlock();
630    }
631    assert serviceBrokers != null;
632    if (!serviceBrokers.isEmpty()) {
633      final Set<Service> allServices = new HashSet<>();
634      try {
635        this.serviceBrokerAssociationLock.writeLock().lock();
636        this.removeServices();
637        final Map<Service, ServiceBroker> associations = new ConcurrentHashMap<>();
638        final Stream<ServiceBroker> serviceBrokerStream;
639        if (this.getParallelServiceDiscovery()) {
640          serviceBrokerStream = serviceBrokers.parallelStream();
641        } else {
642          serviceBrokerStream = serviceBrokers.stream();
643        }        
644        try {
645          serviceBrokerStream
646            .forEach(serviceBroker -> {
647                if (serviceBroker != null) {
648                  Catalog catalog = null;
649                  try {
650                    catalog = this.getCatalog(serviceBroker);
651                  } catch (final ServiceBrokerException serviceBrokerException) {
652                    throw new IllegalStateException(serviceBrokerException);
653                  }
654                  if (catalog != null) {
655                    final Iterable<? extends Service> services = catalog.getServices();
656                    if (services != null) {
657                      for (final Service service : services) {
658                        if (service != null) {
659                          associations.put(service, serviceBroker);
660                        }
661                      }
662                    }
663                  }
664                }
665              });
666        } catch (final IllegalStateException illegalStateException) {
667          final Throwable cause = illegalStateException.getCause();
668          if (cause instanceof ServiceBrokerException) {
669            throw (ServiceBrokerException)cause;
670          }
671          throw illegalStateException;
672        }
673        associations.entrySet().stream()
674          .forEach(entry -> {
675              assert entry != null;
676              final Service service = entry.getKey();
677              assert service != null;
678              final String id = service.getId();
679              if (id != null) {
680                final ServiceBroker serviceBroker = entry.getValue();
681                assert serviceBroker != null;
682                this.putServiceBrokerForServiceId(id, serviceBroker);
683              }
684            });
685      } finally {
