032                                      @NotNull /* @NotEmpty */ final String serviceId,

035  }
036  
037  public DeleteServiceInstanceCommand(final String instanceId,
038                                      @NotNull /* @NotEmpty */ final String serviceId,
039                                      @NotNull /* @NotEmpty */ final String planId,
040                                      final boolean acceptsIncomplete) {
041    super(instanceId, serviceId, planId, acceptsIncomplete);
042    if (!Boolean.getBoolean("org.microbean.servicebroker.api.lenient")) {
043      // The specification requires non-null serviceId and planId
044      // parameters.  However, some clients do not supply them.
045      Objects.requireNonNull(serviceId, () -> "serviceId must not be null");
046      Objects.requireNonNull(planId, () -> "planId must not be null");
047    }
048  }
