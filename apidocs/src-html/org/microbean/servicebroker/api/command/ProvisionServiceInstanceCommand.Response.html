017package org.microbean.servicebroker.api.command;
018
019import java.net.URI;
020
021import java.util.Map;
022import java.util.Objects;
023
024// import javax.validation.constraints.NotEmpty;
025import javax.validation.constraints.NotNull;
026
027public class ProvisionServiceInstanceCommand extends AbstractServiceInstanceMutatingCommand {
028
029  private final String organizationGuid;
030
031  private final String spaceGuid;
032
033  public ProvisionServiceInstanceCommand(final String instanceId,
034                                         @NotNull /* @NotEmpty */ final String serviceId,
035                                         @NotNull /* @NotEmpty */ final String planId,
036                                         final Map<? extends String, ?> parameters,
037                                         @NotNull /* @NotEmpty */ final String organizationGuid,
038                                         @NotNull /* @NotEmpty */ final String spaceGuid) {
039    this(instanceId, serviceId, planId, null, false, organizationGuid, spaceGuid, parameters);
040  }
041  
042  public ProvisionServiceInstanceCommand(final String instanceId,
043                                         @NotNull /* @NotEmpty */ final String serviceId,
044                                         @NotNull /* @NotEmpty */ final String planId,
045                                         final Map<? extends String, ?> parameters,
046                                         final boolean acceptsIncomplete,
047                                         @NotNull /* @NotEmpty */ final String organizationGuid,
048                                         @NotNull /* @NotEmpty */ final String spaceGuid) {
049    this(instanceId, serviceId, planId, null, acceptsIncomplete, organizationGuid, spaceGuid, parameters);
050  }
051  
052  public ProvisionServiceInstanceCommand(final String instanceId,
053                                         @NotNull /* @NotEmpty */ final String serviceId,
054                                         @NotNull /* @NotEmpty */ final String planId,
055                                         final Map<? extends String, ?> context,
056                                         @NotNull /* @NotEmpty */ final String organizationGuid,
057                                         @NotNull /* @NotEmpty */ final String spaceGuid,
058                                         final Map<? extends String, ?> parameters) {
059    this(instanceId, serviceId, planId, context, false, organizationGuid, spaceGuid, parameters);
060  }
061  
062  public ProvisionServiceInstanceCommand(final String instanceId,
063                                         @NotNull /* @NotEmpty */ final String serviceId,
064                                         @NotNull /* @NotEmpty */ final String planId,
065                                         final Map<? extends String, ?> context,
066                                         final boolean acceptsIncomplete,
067                                         @NotNull /* @NotEmpty */ final String organizationGuid,
068                                         @NotNull /* @NotEmpty */ final String spaceGuid,
069                                         final Map<? extends String, ?> parameters) {
070    super(instanceId, serviceId, planId, parameters, acceptsIncomplete);
071    Objects.requireNonNull(planId);
072    Objects.requireNonNull(organizationGuid);
073    Objects.requireNonNull(spaceGuid);
074    this.organizationGuid = organizationGuid;
075    this.spaceGuid = spaceGuid;
076  }
077
078  public final String getOrganizationGuid() {
079    return this.organizationGuid;
080  }
081
082  public final String getSpaceGuid() {
083    return this.spaceGuid;
084  }
085
086  public static class Response extends org.microbean.servicebroker.api.command.AbstractProvisioningResponse {
087
088    private final URI dashboardUri;
089
090    public Response() {
091      super(null);
092      this.dashboardUri = null;
093    }
094    
095    public Response(final URI dashboardUri) {
096      super(null);
097      this.dashboardUri = dashboardUri;
098    }
099
100    public Response(@NotNull /* @NotEmpty */ final String operation) {
101      this(null, operation);
102    }
103    
104    public Response(final URI dashboardUri, @NotNull /* @NotEmpty */ final String operation) {
105      super(operation);
106      this.dashboardUri = dashboardUri;
107    }
108
109    public final URI getDashboardUri() {
110      return this.dashboardUri;
111    }
112    
113  }
114  
115}
